CREATE TABLE `NS_OFFLINE_WX_ORDER` (
  `ORDER_ID` varchar(32) DEFAULT NULL COMMENT '商户单号',
  `MERCH_ID` varchar(11) DEFAULT NULL COMMENT '商户编号',
  `TXN_ID` varchar(32) NOT NULL COMMENT '系统单号',
  `PARTNER` varchar(32) DEFAULT NULL,
  `TUNNEL_ID` varchar(10) DEFAULT NULL COMMENT '通道编码',
  `PMT_TP` varchar(10) DEFAULT NULL COMMENT '支付类型',
  `CCY` varchar(3) DEFAULT NULL COMMENT '币种',
  `AMOUNT` int(11) DEFAULT NULL COMMENT '交易金额',
  `OUT_TXN_ID` varchar(32) DEFAULT NULL COMMENT '发往第三方单号',
  `APP_ID` varchar(64) DEFAULT NULL,
  `APP_TP` char(10) DEFAULT NULL COMMENT '应该类型',
  `USER_ID` varchar(64) DEFAULT NULL COMMENT '用户编号',
  `GOODS_TYPE` varchar(32) DEFAULT NULL COMMENT '商品类型',
  `GOODS_NAME` varchar(255) DEFAULT NULL COMMENT '商品名称',
  `STATUS` varchar(2) DEFAULT NULL COMMENT '交易状态',
  `CHK_STS` varchar(2) DEFAULT NULL COMMENT '核对状态',
  `RET_CODE` varchar(4) DEFAULT NULL COMMENT '响应码',
  `RET_INFO` varchar(128) DEFAULT NULL COMMENT '响应信息',
  `DESCRIPTION` varchar(255) DEFAULT NULL COMMENT '订单描述',
  `RETURN_URL` varchar(512) DEFAULT NULL COMMENT '回执地址',
  `NOTIFY_URL` varchar(512) DEFAULT NULL COMMENT '通知地址',
  `EX_TXN_ID` varchar(64) DEFAULT NULL COMMENT '第三方返回单号',
  `WX_APP_ID` varchar(32) DEFAULT NULL COMMENT '微信应用ID',
  `RECEIVE_TIME` timestamp NULL DEFAULT NULL,
  `TRADE_TIME` timestamp NULL DEFAULT NULL,
  `NOTIFY_TIME` timestamp NULL DEFAULT NULL,
  `COMPLETE_TIME` timestamp NULL DEFAULT NULL,
  `REMARK` varchar(255) DEFAULT NULL COMMENT '备注',
  PRIMARY KEY (`TXN_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='微信刷卡支付订单表';
